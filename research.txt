        )

        cursor = connection.cursor(pymysql.cursors.DictCursor)

        cursor.execute("SELECT * FROM users WHERE email=%s", (email,))
        user = cursor.fetchone()
        connection.close()

        if user and check_password_hash(user["password"], password):
            return jsonify({
                "message": "User logged in successfully",
                "user": {
                    "user_id": user["user_id"],
                    "username": user.get("username"),
                    "email": user["email"],
                    "is_admin": user.get("is_admin", 0)
                }
            })
        else:
            return jsonify({"message": "Login failed"}), 401

@app.route("/api/request-reset", methods=["POST"])
def request_reset():
    email = request.json.get("email")
    if not email:
        return jsonify({"success": False, "message": "Email required"}), 400

    # Connect to DB
    connection = pymysql.connect(host="localhost", user="root", password="", database="sokogardenonline")
    cursor = connection.cursor(pymysql.cursors.DictCursor)

    # Check if user exists
    cursor.execute("SELECT * FROM users WHERE email=%s", (email,))
    user = cursor.fetchone()
    if not user:
        connection.close()
        return jsonify({"success": False, "message": "Email not found"}), 404

    # Generate reset token
    import string, random
    from datetime import datetime, timedelta

    token = ''.join(random.choices(string.ascii_letters + string.digits, k=32))
    expiry = datetime.now() + timedelta(hours=1)

    # Create password_resets table if not exists
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS password_resets (
            email VARCHAR(255),
            token VARCHAR(255),
            expiry DATETIME
        )
    """)

    # Remove any old tokens for this user
    cursor.execute("DELETE FROM password_resets WHERE email=%s", (email,))
    # Insert new token
    cursor.execute("INSERT INTO password_resets(email, token, expiry) VALUES (%s, %s, %s)", (email, token, expiry))
    connection.commit()
    connection.close()

    # TODO: Send email with link to frontend reset page
    print(f"Password reset link (for testing): http://localhost:3000/reset-password/{token}")

    return jsonify({"success": True, "message": "Reset link sent to your email"})

from werkzeug.security import generate_password_hash

@app.route("/api/reset-password/<token>", methods=["POST"])
def reset_password(token):
    new_password = request.json.get("password")
    if not new_password:
        return jsonify({"success": False, "message": "Password is required"}), 400

    connection = pymysql.connect(host="localhost", user="root", password="", database="sokogardenonline")
    cursor = connection.cursor(pymysql.cursors.DictCursor)

    # Check if token is valid
    cursor.execute("SELECT * FROM password_resets WHERE token=%s", (token,))
    record = cursor.fetchone()
    if not record or datetime.now() > record["expiry"]:
        connection.close()
        return jsonify({"success": False, "message": "Invalid or expired token"}), 400

    # Update user password
    hashed_password = generate_password_hash(new_password)
    cursor.execute("UPDATE users SET password=%s WHERE email=%s", (hashed_password, record["email"]))

    # Remove token after use
    cursor.execute("DELETE FROM password_resets WHERE token=%s", (token,))
    connection.commit()
    connection.close()

    return jsonify({"success": True, "message": "Password updated successfully"})
